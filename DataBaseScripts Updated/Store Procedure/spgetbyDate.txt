-- FUNCTION: pay.spgetbydatemaster(timestamp without time zone, refcursor)

-- DROP FUNCTION IF EXISTS pay.spgetbydatemaster(timestamp without time zone, refcursor);

CREATE OR REPLACE FUNCTION pay.spgetbydatemaster(
	a_id timestamp without time zone,
	ref1 refcursor)
    RETURNS refcursor
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
AS $BODY$
BEGIN
	OPEN ref1 for
		select * from pay.master where  p_date = a_id;
	return ref1;
END;
$BODY$;

ALTER FUNCTION pay.spgetbydatemaster(timestamp without time zone, refcursor)
    OWNER TO postgres;
